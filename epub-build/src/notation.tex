\chapter*{Notation}
\addcontentsline{toc}{chapter}{Notation}

This book uses consistent notation throughout. Below is a reference guide for the symbols and conventions used.

\section*{Sets and Number Systems}

\begin{tabular}{@{}ll@{}}
$\R$ & Real numbers \\
$\R^n$ & $n$-dimensional real vectors \\
$\R_+$, $\R^n_+$ & Non-negative reals / non-negative real vectors \\
$\Z$ & Integers \\
$\Z_+$, $\Z^n_+$ & Non-negative integers / non-negative integer vectors \\
$\Q$ & Rational numbers \\
$\{0,1\}^n$ & Binary vectors of dimension $n$ \\
\end{tabular}

\section*{Vectors and Matrices}

\begin{tabular}{@{}ll@{}}
$\x, \y, \z$ & Decision variable vectors (bold lowercase) \\
$\a, \b, \c$ & Parameter vectors (bold lowercase) \\
$\0$ & Zero vector \\
$A, B$ & Matrices (capital italic) \\
$x_i$ & $i$-th component of vector $\x$ \\
$a_{ij}$ & Element in row $i$, column $j$ of matrix $A$ \\
$\x^\top$ & Transpose of vector $\x$ \\
$A^\top$ & Transpose of matrix $A$ \\
$\c^\top \x$ & Dot product (inner product) of $\c$ and $\x$ \\
$\|\x\|$ & Norm of vector $\x$ \\
\end{tabular}

\section*{Optimization Problems}

A standard linear program is written as:
\[
\begin{split}
\max \quad & \c^\top \x \\
\st \quad & A\x \leq \b \\
& \x \geq \0
\end{split}
\]

\begin{tabular}{@{}ll@{}}
$\max$, $\min$ & Maximize, minimize \\
s.t. & Subject to (constraints follow) \\
$\leq$, $\geq$, $=$ & Constraint types \\
$\x^*$ & Optimal solution \\
$z^*$ & Optimal objective value \\
\end{tabular}

\section*{Summation and Indexing}

\begin{tabular}{@{}ll@{}}
$\sum_{i=1}^{n} x_i$ & Sum of $x_1 + x_2 + \cdots + x_n$ \\
$\sum_{i \in I} x_i$ & Sum over all indices in set $I$ \\
$\prod_{i=1}^{n} x_i$ & Product $x_1 \cdot x_2 \cdots x_n$ \\
$\forall i \in I$ & For all $i$ in set $I$ \\
$\exists$ & There exists \\
\end{tabular}

\section*{Functions and Operators}

\begin{tabular}{@{}ll@{}}
$f(x)$ & Function of $x$ \\
$\nabla f$ & Gradient of $f$ \\
$\nabla^2 f$ & Hessian matrix of $f$ \\
$\argmin_x f(x)$ & Value of $x$ that minimizes $f$ \\
$\conv(S)$ & Convex hull of set $S$ \\
$\dom(f)$ & Domain of function $f$ \\
\end{tabular}

\section*{Complexity Classes}

\begin{tabular}{@{}ll@{}}
\polynomial & Polynomial-time solvable \\
\npcomplete & NP-Complete problem \\
\nphard & NP-Hard problem \\
\end{tabular}
