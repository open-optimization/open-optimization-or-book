% ---- layout parameters ----
\def\Yh{0.35}            % half bar height

% Plot frame and grid

\node[left] at (\xmin-0.5,\yMone)  {Machine 1};
\node[left] at (\xmin-0.5,\yMzero) {Machine 0};

% ---- Gantt bars (aligned exactly to the times) ----
% Machine 2
\gbar{0}{2}{\yMtwo}{job1}
