\documentclass[tikz,border=3.14mm]{standalone}
\begin{document}
\begin{tikzpicture}

    % Highlight the feasible region
    \fill[blue!20,opacity=0.7] 
        (0,20) -- (0,160) -- (60,220) -- (100,180) -- (160,120) -- (160,20) -- (0,20) -- cycle;

    % Define axes
    \draw[thick,->] (0,0) -- (170,0) node[anchor=north] {$r$};
    \draw[thick,->] (0,0) -- (0,330) node[anchor=east] {$t$};

    % Add tick marks on the r-axis
    \foreach \x in {20,40,60,80,100,120,140,160} {
        \draw (\x,0.1) -- (\x,-0.1) node[anchor=north] {\scriptsize $\x$};
    }

    % Add tick marks on the t-axis
    \foreach \y in {50,100,150,200,250,300} {
        \draw (0.1,\y) -- (-0.1,\y) node[anchor=east] {\scriptsize $\y$};
    }

    % Define the constraints
    % Line 1: 2r + t = 320
    \draw[red,thick] (0,320) -- (160,0);
    \node[red, rotate=-45, anchor=north west] at (80,160) {$2r + t \leq 320$};

    % Line 2: r = 160
    \draw[green!50!black,thick] (160,0) -- (160,320);
    \node[green!50!black, rotate=90, anchor=south] at (160,150) {$r \leq 160$};

    % Line 3: t = 320
    \draw[purple,thick] (0,320) -- (170,320);
    \node[purple, rotate=0, anchor=south] at (80,320) {$t \leq 320$};

    % Line 4: 5r + t = 500
    \draw[orange,thick] (0,500) -- (100,0);
    \node[orange, rotate=-75, anchor=north west] at (50,200) {$5r + t \leq 500$};

    % Line 5: r + t = 280
    \draw[blue!50!black,thick] (0,280) -- (160,120);
    \node[blue!50!black, rotate=-45, anchor=north west] at (80,200) {$r + t \leq 280$};

    % Line 6: t = 20
    \draw[cyan,thick] (0,20) -- (170,20);
    \node[cyan, rotate=0, anchor=north] at (80,20) {$t \geq 20$};

    % Label feasible region
    \node[blue!70!black, font=\bfseries] at (80,100) {Feasible Region};

\end{tikzpicture}
\end{document}
