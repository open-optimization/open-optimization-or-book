\makeatother

\providecommand{\citeauthor}[1]{[Author]}
\providecommand{\citedate}[1]{[Date]}
\providecommand{\citeurl}[1]{[URL]}

% Additional biblatex citation commands used in content
\providecommand{\parencite}[2][]{[#2]}
\providecommand{\textcite}[2][]{[#2]}
\providecommand{\autocite}[2][]{[#2]}
\providecommand{\fullcite}[1]{[Full citation: #1]}
\providecommand{\footfullcite}[1]{\footnote{[#1]}}
\providecommand{\citeyear}[1]{[Year]}
\providecommand{\citenum}[1]{[#1]}
\providecommand{\volcite}[3][]{[#2, #3]}

% Dummy print bibliography
\providecommand{\printbibliography}[1][]{}

% Dummy refsection environment
\newenvironment{refsection}[1][]{}{}

\providecommand{\addbibresource}[1]{}
