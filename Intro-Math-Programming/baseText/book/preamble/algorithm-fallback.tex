% =============================================================================
% ALGORITHM FALLBACK DEFINITIONS
% =============================================================================
% This file provides fallback definitions for algorithm environments when
% the algorithmicx package is not available.
% =============================================================================

% Algorithm environment (float-like)
\newenvironment{algorithm}[1][]{%
    \begin{center}\begin{minipage}{0.9\textwidth}\hrule\medskip\textbf{Algorithm}\medskip
}{%
    \medskip\hrule\end{minipage}\end{center}
}

% Algorithmic environment (for pseudocode)
\newenvironment{algorithmic}[1][]{%
    \begin{quote}\ttfamily
}{%
    \end{quote}
}

% Basic commands
\newcommand{\State}{\par\noindent}
\newcommand{\Statex}{\par\noindent}

% Control flow
\newcommand{\If}[1]{\textbf{if} #1 \textbf{then}}
\newcommand{\ElsIf}[1]{\textbf{else if} #1 \textbf{then}}
\newcommand{\Else}{\textbf{else}}
\newcommand{\EndIf}{\textbf{end if}}
\newcommand{\For}[1]{\textbf{for} #1 \textbf{do}}
\newcommand{\EndFor}{\textbf{end for}}
\newcommand{\ForAll}[1]{\textbf{for all} #1 \textbf{do}}
\newcommand{\While}[1]{\textbf{while} #1 \textbf{do}}
\newcommand{\EndWhile}{\textbf{end while}}
\newcommand{\Repeat}{\textbf{repeat}}
\newcommand{\Until}[1]{\textbf{until} #1}
\newcommand{\Loop}{\textbf{loop}}
\newcommand{\EndLoop}{\textbf{end loop}}

% Return and I/O
\newcommand{\Return}[1]{\textbf{return} #1}
\newcommand{\Require}[1]{\textbf{Require:} #1}
\newcommand{\Ensure}[1]{\textbf{Ensure:} #1}
\newcommand{\Print}[1]{\textbf{print} #1}

% Functions and procedures
\newcommand{\Function}[2]{\textbf{function} \textsc{#1}(#2)}
\newcommand{\EndFunction}{\textbf{end function}}
\newcommand{\Procedure}[2]{\textbf{procedure} \textsc{#1}(#2)}
\newcommand{\EndProcedure}{\textbf{end procedure}}
\newcommand{\Call}[2]{\textsc{#1}(#2)}

% Comments
\newcommand{\Comment}[1]{\hfill\(\triangleright\) \textit{#1}}
\newcommand{\LineComment}[1]{\par\noindent\(\triangleright\) #1}

% Algorithm substate environment
\newcounter{algsubstate}
\newenvironment{algsubstates}{}{}
