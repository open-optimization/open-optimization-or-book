%% ============================================================
%% Centerpiece: layered TikZ composition
%% ============================================================
\begin{tikzpicture}[
    scale=0.95,
    every node/.style={font=\sffamily\small},
  ]

  %% Soft background tint (very subtle) — sized to enclose all three motifs
  \fill[blue!2] (-6.6,-0.9) rectangle (6.6,3.6);

  %% ---------- LEFT: LP feasible region (Part I motif) ----------
  \begin{scope}[shift={(-5.7, 0)}]
    % axes
    \draw[->, gray!60, thick] (-0.4,0) -- (3.2,0) node[right, font=\scriptsize\sffamily, gray!70] {$x_1$};
    \draw[->, gray!60, thick] (0,-0.4) -- (0,3.0) node[above, font=\scriptsize\sffamily, gray!70] {$x_2$};
    % feasible region: pentagon with five vertices
    \coordinate (A) at (0,0);
    \coordinate (B) at (2.6,0);
    \coordinate (C) at (2.6,1.0);
    \coordinate (D) at (1.5,2.2);
    \coordinate (E) at (0,2.4);
    \fill[blue!18] (A) -- (B) -- (C) -- (D) -- (E) -- cycle;
    \draw[blue!70!black, thick] (A) -- (B) -- (C) -- (D) -- (E) -- cycle;
    % parallel level sets of the objective (dashed)
    \draw[orange!80!black, dashed, thin]  (-0.2, 0.6) -- (2.2,-0.4);
    \draw[orange!80!black, dashed, thin]  (-0.2, 1.6) -- (3.0, 0.0);
    \draw[orange!80!black, dashed, thick] (-0.2, 2.6) -- (3.4, 0.8);
    % optimum at D
    \fill[orange!90!red] (D) circle (3.2pt);
    \node[font=\scriptsize\sffamily, above right=-1pt of D, orange!60!black] {$\mathbf{x}^\star$};
  \end{scope}

  %% ---------- CENTER: small graph / network (Part II motif) ----------
  \begin{scope}[shift={(0, 1.2)}, every node/.style={circle, draw=teal!70!black, fill=teal!12, minimum size=4.5mm, inner sep=0pt, font=\scriptsize\sffamily}]
    \node (n1) at ( 0.0,  1.2) {};
    \node (n2) at (-1.2,  0.4) {};
    \node (n3) at ( 1.2,  0.4) {};
    \node (n4) at (-0.7, -0.9) {};
    \node (n5) at ( 0.7, -0.9) {};
    \node (n6) at ( 0.0, -1.7) {};
    % all edges in pale teal
    \begin{scope}[every edge/.style={draw=teal!50, thick}]
      \path (n1) edge (n2)
            (n1) edge (n3)
            (n2) edge (n4)
            (n3) edge (n5)
            (n4) edge (n5)
            (n4) edge (n6)
            (n5) edge (n6);
    \end{scope}
    % highlighted "shortest path" in darker teal
    \begin{scope}[every edge/.style={draw=teal!85!black, line width=1.6pt}]
      \path (n1) edge (n2)
            (n2) edge (n4)
            (n4) edge (n6);
    \end{scope}
    \node[draw=none, fill=none, font=\scriptsize\sffamily\itshape, gray!70] at (0, 1.95) {graph};
  \end{scope}

  %% ---------- RIGHT: integer-programming lattice (Part III motif) ----------
  \begin{scope}[shift={(2.7, 0)}]
    \draw[->, gray!60, thick] (-0.4,0) -- (3.2,0) node[right, font=\scriptsize\sffamily, gray!70] {$x_1$};
    \draw[->, gray!60, thick] (0,-0.4) -- (0,3.0) node[above, font=\scriptsize\sffamily, gray!70] {$x_2$};
    % LP relaxation region (light gray)
    \fill[gray!10] (0,0) -- (2.8,0) -- (2.8,0.9) -- (1.3,2.4) -- (0,2.4) -- cycle;
    \draw[gray!45, thick] (0,0) -- (2.8,0) -- (2.8,0.9) -- (1.3,2.4) -- (0,2.4) -- cycle;
    % integer lattice points: blue dots inside feasible, hollow outside
    \foreach \x in {0,1,2}
      \foreach \y in {0,1,2}{
        \fill[purple!75!black] (\x,\y) circle (2.0pt);
      }
    % a few outside the region: hollow
    \draw[purple!75!black] (3,0) circle (2.0pt);
    \draw[purple!75!black] (3,1) circle (2.0pt);
    \draw[purple!75!black] (2,3) circle (2.0pt);
    \fill[purple!75!black, opacity=0.6] (1,2) circle (3.0pt);  % integer optimum
  \end{scope}

\end{tikzpicture}
